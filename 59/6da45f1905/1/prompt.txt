### `async query()` function
```python
async def query(query_input_path, index_dir, collection_name, output_dir, max_concurrency):
    global vector_store
    vector_store = Chroma(persist_directory=index_dir, embedding_function=get_embeddings(),
                          create_collection_if_not_exists=False, collection_name=collection_name)
    graph = StateGraph(State).add_sequence([retrieve, generate])
    graph.add_edge(START, "retrieve")
    graph = graph.compile()
    qa_pairs = json.load(open(query_input_path))
    outputs = await graph.abatch([{"question": d["query"]} for d in qa_pairs],
                                  config=RunnableConfig(max_concurrency=max_concurrency))
    results = [dict(doc, response={"content": out["answer"],
               "contexts": [d.page_content for d in out["context"]]})
               for doc, out in zip(qa_pairs, outputs)]
    out_path = f"{output_dir}/response.json"
    Path(out_path).parent.mkdir(parents=True, exist_ok=True)
    json.dump(results, open(out_path, "w"), indent=4, ensure_ascii=False)
```
